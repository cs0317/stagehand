@dataclass
class MonsterRequest:
    query: str
    location: str
    max_results: int

@dataclass
class JobItem:
    title: str
    company: str
    location: str
    salary: str
    date: str

@dataclass
class MonsterResult:
    jobs: List[JobItem]
