# Searches TripAdvisor for restaurants in a given city (via Google redirect)
# and extracts up to max_results restaurants with name, cuisine type, rating,
# and price level.

@dataclass(frozen=True)
class TripAdvisorRestaurantSearchRequest:
    destination: str
    max_results: int

@dataclass(frozen=True)
class TripAdvisorRestaurant:
    name: str
    cuisine: str
    rating: str
    price_level: str
