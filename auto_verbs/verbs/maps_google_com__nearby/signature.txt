class NearbySearchRequest:
    query: str          # search term, e.g. "dealerships"
    location: str       # location context, e.g. "urbana champaign"
    max_results: int = 5  # maximum number of businesses to return


@dataclass(frozen=True)
class BusinessDetail:
    name: str       # business name
    address: str    # street address
    rating: str     # star rating, e.g. "4.5"
    phone: str      # phone number
    website: str    # website URL

class NearbySearchResult:

    location: str
    businesses: List[BusinessDetail]
