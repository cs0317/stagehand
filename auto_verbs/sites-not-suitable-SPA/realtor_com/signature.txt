@dataclass
class RealtorRequest:
    location: str
    min_price: int
    max_price: int
    max_results: int

@dataclass
class ListingItem:
    address: str
    price: str
    bedrooms: str
    bathrooms: str
    sqft: str
    status: str

@dataclass
class RealtorResult:
    listings: List[ListingItem]
