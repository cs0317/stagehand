@dataclass(frozen=True)
class WhiskyadvocateSearchRequest:
    search_query: str = "bourbon"
    max_results: int = 5

    distillery: str = ""
    region: str = ""
    score: str = ""
    price: str = ""
    age: str = ""
    proof: str = ""
    review_summary: str = ""


@dataclass
class WhiskyadvocateSearchResult:
